function Feed()
    spawn(function()
        _G.Feed = true
        while _G.Feed do
            wait()
            pcall(function()
                function eat()
                    local v54 = game:GetService("Players").LocalPlayer
                    local v55 = v54:WaitForChild("Backpack")
                    local v56, v57, v58 = ipairs(v55:GetChildren())
                    while true do
                        local v59
                        v58, v59 = v56(v57, v58)
                        if v58 == nil then
                            break
                        end
                        if v59:IsA("Tool") and v59:GetAttribute("ItemType") == "Holdable" then
                            v59.Parent = v54.Character
                            break
                        end
                    end
                end
                local v60, v61, v62 = pairs(workspace.PetsPhysical:GetChildren())
                while true do
                    local v63
                    v62, v63 = v60(v61, v62)
                    if v62 == nil then
                        break
                    end
                    if v63:GetAttribute("OWNER") == game:GetService("Players").LocalPlayer.Name then
                        eat()
                        local v64 = {
                            "Feed",
                            v63:GetAttribute("UUID")
                        }
                        game:GetService("ReplicatedStorage").GameEvents.ActivePetService:FireServer(unpack(v64))
                        wait(1)
                    end
                end
                wait(5)
            end)
        end
    end)
end
