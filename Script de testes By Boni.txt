-- FunÃ§Ã£o para pegar lista de jogadores
local function getPlayerList()
    local players = {""}
    for _, player in pairs(game.Players:GetPlayers()) do
        if player ~= game.Players.LocalPlayer then
            table.insert(players, player.Name)
        end
    end
    return players
end
